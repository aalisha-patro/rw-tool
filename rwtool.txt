const findParentFolders = (folderId, acc = []) => {
  const folder = folderData.find((folder) => folder.id === folderId);
  if (folder) {
    acc.unshift(folder.name);
    if (folder.parentId) {
      findParentFolders(folder.parentId, acc);
    }
  }
  return acc;
};

const handleFileClick = (file) => {
  const MAX_RECENT_FILES = 5;

  const folder = folderData.find((folder) => folder.id === file.folderIds[0]);
  const parentFolders = findParentFolders(folder.id, []);
  const path = [...parentFolders, folder.name, file.name].join("/");

  const updatedClickedFileBreadcrumbs = {
    [file.id]: path,
  };

  setClickedFileBreadcrumbs(updatedClickedFileBreadcrumbs);

  // Rest of your code
};
In this code, the findParentFolders function recursively traverses through the parent folders of a given folder until it reaches the root folder. It builds an array of folder names and then joins them with "/" to create the breadcrumb path. This way, the breadcrumb path will include the names of all parent folders leading to the clicked file.
